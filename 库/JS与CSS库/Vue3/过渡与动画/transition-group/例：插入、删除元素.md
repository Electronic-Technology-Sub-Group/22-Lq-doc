# 例：插入、删除元素

```html
<div id="list-app">
    <button @click="add">随机添加元素</button>&nbsp;&nbsp;
    <button @click="del">随机删除元素</button>

    <transition-group name="list" tag="p">
        <span v-for="item in items" :key="item" 
              class="list-item">{{item}}</span>
    </transition-group>
</div>
```

```css
.list-item {
    display: inline-block;
    margin-right: 10px;
}
.list-enter-active,
.list-leave-active {
    transition: all 0.5s;
}
.list-enter-from,
.list-leave-to {
    opacity: 0;
    transform: translateY(30px);
}
```

```js
Vue.createApp({
    data() {
        return {
            // 展示的数据列表
            items: [1, 2, 3, 4, 5, 6, 7, 8, 9],
            // 下一个元素
            // key 绑定在 item，需要保证插入的数据是唯一的
            next: 10
        }
    },
    methods: {
        randomIndex() {
            return Math.floor(Math.random() * this.items.length)
        },
        add() {
            this.items.splice(this.randomIndex(), 0, this.next++)
        },
        del() {
            this.items.splice(this.randomIndex(), 1)
        }
    }
}).mount('#list-app')
```

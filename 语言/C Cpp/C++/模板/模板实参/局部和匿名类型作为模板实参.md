允许局部和匿名类型作为模板实参

#cpp11 

```cpp
template <class T> class X {};
template <class T> void f(T t) {};

// 匿名类型
struct {} unnamed_obj;

int main() {
    // 局部类型
    struct A {};
    typedef struct {} B;
    // 局部匿名类型
    enum { e };

    X<A>  x1;
    X<A*> x2;
    X<B>  x3;
    f(e);
    f(unnamed_obj);
    B b;
    f(b);
    return 0;
}
```
